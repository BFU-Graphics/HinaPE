cmake_minimum_required(VERSION 3.16)
project(HINAPE_FLUID_EXPERIMENTAL VERSION 1.21 DESCRIPTION "HinaPE is not a Physics Engine" LANGUAGES CXX)

set(HINAPE_COMMON_DIR "../../common")
set(HINAPE_DEPS_DIR "../deps")

if (NOT TARGET HinaPE_Common)
    add_subdirectory(${HINAPE_COMMON_DIR} HinaPE_Common)
endif ()

file(GLOB_RECURSE HINAPE_FLUID_EXPERIMENTAL_SRC *.h *.cpp *.inl.h)
add_library(HINAPE_FLUID_EXPERIMENTAL ${HINAPE_FLUID_EXPERIMENTAL_SRC})
set_target_properties(HINAPE_FLUID_EXPERIMENTAL PROPERTIES CXX_STANDARD 20 CXX_EXTENSIONS OFF)
target_include_directories(HINAPE_FLUID_EXPERIMENTAL PUBLIC ${HINAPE_COMMON_DIR} ${HINAPE_DEPS_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../)
target_include_directories(HINAPE_FLUID_EXPERIMENTAL PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})
target_link_libraries(HINAPE_FLUID_EXPERIMENTAL PUBLIC HinaPE_Common)
